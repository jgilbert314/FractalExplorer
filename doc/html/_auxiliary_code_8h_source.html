34 
35  // Throw an error if the interval is improperly defined
36  if (bound_L >= bound_U) {
37  throw runtime_error( "Upper bound must be greater than lower bound." );
38  }
39  if (N <= 1) {
40  throw invalid_argument( "The number of points must be greater than 1." );
41  }
42 
43  // Distance between points in set
44  double dist = (bound_U - bound_L) / (N - 1);
45  // Calculate points in set
46  std::vector<double> output(N);
47  for (int itr = 0; itr < N; itr++) {
48  // x_n = x_0 + n*(x_f - x_0)/(N - 1)
49  output[itr] = bound_L + itr*dist;
50  }
51
